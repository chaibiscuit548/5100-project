# Prepare data: ensure POST_ELECTION_PUBLIC_TRUST is numeric (1-5) and AGE is ordered factor
df <- df %>%
  mutate(
    POST_ELECTION_PUBLIC_TRUST = as.numeric(as.character(POST_ELECTION_PUBLIC_TRUST)),   # adjust if already numeric
    AGE = factor(AGE, levels = c("18-24","25-34","35-44","45-54","55-64","65+"))
  )

# Compute mean and 95% CI by age
summary_df <- df %>%
  group_by(AGE) %>%
  summarize(
    n = sum(!is.na(POST_ELECTION_PUBLIC_TRUST)),
    mean_trust = mean(POST_ELECTION_PUBLIC_TRUST, na.rm = TRUE),
    sd_trust = sd(POST_ELECTION_PUBLIC_TRUST, na.rm = TRUE)
  ) %>%
  mutate(
    se = sd_trust / sqrt(n),
    ci_lower = mean_trust - qt(0.975, df = pmax(n - 1, 1)) * se,
    ci_upper = mean_trust + qt(0.975, df = pmax(n - 1, 1)) * se
  )

# Plot
ggplot(summary_df, aes(x = AGE, y = mean_trust, group = 1)) +
  geom_line(color = "#2C7FB8", size = 1) +
  geom_point(size = 3, color = "#2C7FB8") +
  geom_ribbon(aes(ymin = ci_lower, ymax = ci_upper), alpha = 0.15, fill = "#2C7FB8") +
  labs(
    x = "Age group",
    y = "Mean Public Trust Post Election",
    title = "Mean Public Trust Post Election by Age",
    subtitle = "Points = group means; ribbon = 95% CI"
  ) +
  theme_minimal(base_size = 14)

# Prepare data: ensure POST ELECTION_LOCAL_GOVT_TRUST is numeric (1-5) and AGE is ordered factor

df <- df %>%
  mutate(
    POST_ELECTION_LOCAL_GOVT_TRUST = as.numeric(as.character(POST_ELECTION_LOCAL_GOVT_TRUST)),   # adjust if already numeric
    AGE = factor(AGE, levels = c("18-24","25-34","35-44","45-54","55-64","65+"))
  )

# Compute mean and 95% CI by age
summary_df <- df %>%
  group_by(AGE) %>%
  summarize(
    n = sum(!is.na(POST_ELECTION_LOCAL_GOVT_TRUST)),
    mean_trust = mean(POST_ELECTION_LOCAL_GOVT_TRUST, na.rm = TRUE),
    sd_trust = sd(POST_ELECTION_LOCAL_GOVT_TRUST, na.rm = TRUE)
  ) %>%
  mutate(
    se = sd_trust / sqrt(n),
    ci_lower = mean_trust - qt(0.975, df = pmax(n - 1, 1)) * se,
    ci_upper = mean_trust + qt(0.975, df = pmax(n - 1, 1)) * se
  )

# Plot
ggplot(summary_df, aes(x = AGE, y = mean_trust, group = 1)) +
  geom_line(color = "#2C7FB8", size = 1) +
  geom_point(size = 3, color = "#2C7FB8") +
